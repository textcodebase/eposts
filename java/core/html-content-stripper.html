The Following code sample stripes content from a HTML file.

HTMLStripper.java 
        
                            
                                package com.text.eposts.io;

                                import java.io.BufferedReader;
                                import java.io.File;
                                import java.io.FileInputStream;
                                import java.io.InputStream;
                                import java.io.InputStreamReader;
                                import java.net.URL;
                                import java.net.URLConnection;

                                public class HTMLStripper {

                                    private String htmlContent = "";
                                    
                                    public HTMLStripper(String httpUrl) throws Exception {
                                        this.htmlContent = getContent(httpUrl);
                                    }
                                    
                                    public HTMLStripper(File htmlFile) throws Exception {
                                        this.htmlContent = getContent(htmlFile);
                                    }
                                    
                                    private String getContent(String strUrl) throws Exception {
                                        URL url = new URL(strUrl);
                                        URLConnection con = url.openConnection();
                                        //int len = con.getContentLength();
                                        //System.out.println(len);
                                        InputStream iStream = con.getInputStream();
                                        BufferedReader reader = new BufferedReader(new InputStreamReader(iStream));
                                        String line = "";
                                        StringBuffer buffer = new StringBuffer();
                                        while((line = reader.readLine())!=null) {
                                            buffer.append(line+"\r\n");
                                        }
                                        String htmlContent = buffer.toString();
                                        return htmlContent;
                                    }
                                    
                                    private String getContent(File file) throws Exception {
                                        FileInputStream iStream = new FileInputStream(file);
                                        byte[] bytes = new byte[iStream.available()];
                                        iStream.read(bytes);
                                        String htmlContent = new String(bytes);
                                        iStream.close();
                                        return htmlContent;
                                    }
                                    
                                    public String strip() {
                                        return this.htmlContent;
                                    }
                                    
                                    
                                    public static void main(String[] args)throws Exception {
                                        HTMLStripper stripper = new HTMLStripper("https://www.w3schools.com/java/default.asp");
                                        String htmlString = stripper.strip(); 
                                //		String htmlString = "content";
                                        
                                        String regexp = "<[^>]*>";
                                //		Pattern pattern = Pattern.compile(regexp);
                                //		Matcher matcher = pattern.matcher(htmlString);
                                        
                                //		System.out.println(matcher.matches());
                                //		htmlString = htmlString.replace
                                        System.out.println(htmlString.replaceAll(regexp, ""));
                                    }

                                }
